        // ---- update all spreadsheet calculations ----

        function updateSpreadsheet() {

            let totalLabor = 0;

            for (let i = 0; i < 10; i++) {

                let totalHours = 0;

                for (let t = 0; t < 5; t++) {

                    const hourInput = document.querySelector(`input[data-staff="${i}"][data-task="${t}"]`);

                    if (hourInput) totalHours += parseFloat(hourInput.value) || 0;

                }

                const hoursCell = document.getElementById(`totalHours-${i}`);

                if (hoursCell) hoursCell.innerText = totalHours;

                const rateInput = document.querySelector(`input[data-staff="${i}"][data-field="rate"]`);

                const rate = parseFloat(rateInput?.value) || 0;

                const laborCost = totalHours * rate;

                totalLabor += laborCost;

                const costCell = document.getElementById(`laborCost-${i}`);

                if (costCell) costCell.innerText = `$${laborCost.toFixed(2)}`;

            }

            totalLaborSpan.innerText = `$${totalLabor.toFixed(2)}`;

            let totalDirect = 0;

            for (let t = 0; t < 5; t++) {

                const equip = parseFloat(document.querySelector(`input[data-task="${t}"][data-cat="equip"]`)?.value) || 0;

                const supplies = parseFloat(document.querySelector(`input[data-task="${t}"][data-cat="supplies"]`)?.value) || 0;

                const permits = parseFloat(document.querySelector(`input[data-task="${t}"][data-cat="permits"]`)?.value) || 0;

                const travel = parseFloat(document.querySelector(`input[data-task="${t}"][data-cat="travel"]`)?.value) || 0;

                const other = parseFloat(document.querySelector(`input[data-task="${t}"][data-cat="other"]`)?.value) || 0;

                const taskTotal = equip + supplies + permits + travel + other;

                totalDirect += taskTotal;

                const taskCell = document.getElementById(`directTaskTotal-${t}`);

                if (taskCell) taskCell.innerText = `$${taskTotal.toFixed(2)}`;

            }

            totalDirectSpan.innerText = `$${totalDirect.toFixed(2)}`;

            grandTotalSpan.innerText = `💰 GRAND TOTAL: $${(totalLabor + totalDirect).toFixed(2)}`;

        }

        // ---- rate builder logic (active graphic) ----

        const baseWageInput = document.getElementById('baseWage');

        const fringePctInput = document.getElementById('fringePct');

        const overheadPctInput = document.getElementById('overheadPct');

        const profitPctInput = document.getElementById('profitPct');

        const segBase = document.getElementById('segBase');

        const segFringe = document.getElementById('segFringe');

        const segOverhead = document.getElementById('segOverhead');

        const segProfit = document.getElementById('segProfit');

        const totalBurdenedSpan = document.getElementById('totalBurdenedRate');

        function updateRateBuilder() {

            const base = parseFloat(baseWageInput.value) || 0;

            const fringePct = parseFloat(fringePctInput.value) || 0;

            const overheadPct = parseFloat(overheadPctInput.value) || 0;

            const profitPct = parseFloat(profitPctInput.value) || 0;

            const fringeAmt = base * (fringePct / 100);

            const laborPlusFringe = base + fringeAmt;

            const overheadAmt = laborPlusFringe * (overheadPct / 100);

            const costPlusOverhead = laborPlusFringe + overheadAmt;

            const profitAmt = costPlusOverhead * (profitPct / 100);

            const total = costPlusOverhead + profitAmt;

            // update segment widths as % of total

            if (total > 0) {

                segBase.style.width = (base / total * 100) + '%';

                segBase.innerText = '$' + base.toFixed(2);

                segFringe.style.width = (fringeAmt / total * 100) + '%';

                segFringe.innerText = '$' + fringeAmt.toFixed(2);

                segOverhead.style.width = (overheadAmt / total * 100) + '%';

                segOverhead.innerText = '$' + overheadAmt.toFixed(2);

                segProfit.style.width = (profitAmt / total * 100) + '%';

                segProfit.innerText = '$' + profitAmt.toFixed(2);

            } else {

                segBase.style.width = '0%';

                segFringe.style.width = '0%';

                segOverhead.style.width = '0%';

                segProfit.style.width = '0%';

                segBase.innerText = '';

                segFringe.innerText = '';

                segOverhead.innerText = '';

                segProfit.innerText = '';

            }

            totalBurdenedSpan.innerHTML = `$${total.toFixed(2)} <small>/hr fully burdened</small>`;

        }

        // attach listeners to rate builder inputs

        [baseWageInput, fringePctInput, overheadPctInput, profitPctInput].forEach(inp => {

            inp.addEventListener('input', updateRateBuilder);

        });

        // attach spreadsheet listeners (delegation)

        function attachSpreadsheetEvents() {

            const container = document.querySelector('.presentation');

            container.addEventListener('input', function(e) {

                if (e.target.matches('input[type="number"]') && !e.target.closest('.rate-builder')) {

                    updateSpreadsheet();

                }

            });

        }

        function resetAll() {

            // reset staff inputs

            for (let i = 0; i < 10; i++) {

                const rateInput = document.querySelector(`input[data-staff="${i}"][data-field="rate"]`);

                if (rateInput) rateInput.value = defaultStaff[i].rate.toFixed(2);

                for (let t = 0; t < 5; t++) {

                    const hourInput = document.querySelector(`input[data-staff="${i}"][data-task="${t}"]`);

                    if (hourInput) hourInput.value = defaultHours[i][t];

                }

            }

            // reset direct inputs

            for (let t = 0; t < 5; t++) {

                const cats = ['equip','supplies','permits','travel','other'];

                cats.forEach(cat => {

                    const inp = document.querySelector(`input[data-task="${t}"][data-cat="${cat}"]`);

                    if (inp) inp.value = defaultDirect[t][cat];

                });

            }

            // reset rate builder to default (25,32,28,10)

            baseWageInput.value = '25.00';

            fringePctInput.value = '32.0';

            overheadPctInput.value = '28.0';

            profitPctInput.value = '10.0';

            updateSpreadsheet();

            updateRateBuilder();

        }

        // initial render

        renderStaff();

        renderDirect();

        attachSpreadsheetEvents();

        updateSpreadsheet();

        updateRateBuilder();

        document.getElementById('resetBtn').addEventListener('click', resetAll);

    })();
